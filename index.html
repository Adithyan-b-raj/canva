function verifyToken(token) {
  try {
    const decoded = jwt.verify(
      token,
      process.env.JWT_SECRET
    );
    return { valid: true, data: decoded };
  } catch (err) {
    return { 
      valid: false, 
      error: err.message 
    };
  }
}

    const fingerprint = createFingerprint(packet);
    const threat = this.signatures.match(fingerprint);
    
    if (threat || this.anomalyDetector.isAnomaly(packet)) {
      await this.blockSource(packet.source);
